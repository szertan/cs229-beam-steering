\block{Problem \& Physical System}{
    \textbf{Goal:} Steer EM beam to target angles using tunable plasma rods.
    
    \vspace{0.3em}
    \centering
    \includegraphics[width=0.6\linewidth]{figures/domain_schematic.pdf}
    
    \raggedright
    \vspace{0.3em}
    \textbf{Plasma rods:} $\varepsilon(\rho) = 1 - (\rho \omega_p^{\max}/\omega)^2$
    
    \textbf{Design:} 64 parameters $\boldsymbol{\rho} \in [0,1]^{64}$
    
    \textbf{Simulation:} FDFD, $400\times400$ grid, 6 GHz
}

\block{Methods}{
    \textbf{ES-Single:} One design per angle $\theta^*$
    \begin{equation*}
        R = P_{\theta^*} - 0.5 \sum_{\theta \neq \theta^*} P_\theta
    \end{equation*}
    
    \textbf{ES-Multi:} One design for all angles
    \begin{equation*}
        R = \sum_\theta P_\theta - 0.5 \cdot \text{Var}(P_\theta)
    \end{equation*}
    
    \textbf{ES+NN:} Neural network $f_\phi: [\sin\theta, \cos\theta] \mapsto \boldsymbol{\rho}_{8\times8}$
    \begin{equation*}
        R = \sum_{\theta \in \Theta} \left[ P_\theta - 0.5 \sum_{\theta' \neq \theta} P_{\theta'} \right]
    \end{equation*}
}

\block{ES Algorithm}{
    1. Sample $N=100$ perturbations $\boldsymbol{\epsilon}_i \sim \mathcal{N}(0, I)$
    
    2. Evaluate $R_i = R(\boldsymbol{\theta} + \sigma \boldsymbol{\epsilon}_i)$
    
    3. Gradient: $\hat{\mathbf{g}} = \frac{1}{N\sigma} \sum w_i \boldsymbol{\epsilon}_i$
    
    4. Update with Adam ($\eta=0.02$, $\sigma_0=0.3$)
}

\block{ES+NN Algorithm}{
    1. Sample perturbations $\boldsymbol{\epsilon}_i$ for NN weights $\phi$
    
    2. For each $\theta \in \{0^\circ, 90^\circ, 180^\circ\}$:
    
    \quad Generate design $\boldsymbol{\rho} = f_{\phi + \sigma\boldsymbol{\epsilon}_i}(\theta)$
    
    \quad Simulate and compute $P_\theta$
    
    3. Compute total reward $R_i$ across all angles
    
    4. Update NN weights (20,928 params) with Adam
}

\column{0.25}

\block{Conclusions}{
    \textbf{ES-Single Results:}
    
    \quad $0^\circ$: $P = 1.453\times10^6$, crosstalk 0.003\%
    
    \quad $90^\circ$: $P = 1.452\times10^6$, crosstalk 0.005\%
    
    \quad $180^\circ$: $P = 1.448\times10^6$, crosstalk 0.003\%
    
    \vspace{0.3em}
    $\bullet$ \textbf{ES-Multi} confirms no universal design exists
    
    \vspace{0.3em}
    $\bullet$ \textbf{ES+NN} mode collapse reveals $0^\circ/90^\circ$ compatible, $180^\circ$ incompatible
    
    \vspace{0.3em}
    $\bullet$ Gradient-free approach is \textbf{viable} for inverse EM design
}
